def longest_increasing_subsequence(nums):
    if not nums:
        return []
    
    n = len(nums)
    lengths = [1] * n
    previous_indices = [-1] * n
    max_length_index = 0

    for i in range(1, n):
        for j in range(i):
            if nums[i] > nums[j] and lengths[i] < lengths[j] + 1:
                lengths[i] = lengths[j] + 1
                previous_indices[i] = j
        if lengths[i] > lengths[max_length_index]:
            max_length_index = i

    lis = []
    while max_length_index != -1:
        lis.append(nums[max_length_index])
        max_length_index = previous_indices[max_length_index]

    return lis[::-1]
