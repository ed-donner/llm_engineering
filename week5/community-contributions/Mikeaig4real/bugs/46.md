def ford_fulkerson(capacity, source, sink):
    def bfs(parent):
        visited = [False] * len(capacity)
        queue = [source]
        visited[source] = True
        while queue:
            u = queue.pop(0)
            for v, cap in enumerate(capacity[u]):
                if not visited[v] and cap > 0:
                    queue.append(v)
                    visited[v] = True
                    parent[v] = u
                    if v == sink:
                        return True
        return False

    parent = [-1] * len(capacity)
    max_flow = 0
    while bfs(parent):
        path_flow = float('Inf')
        s = sink
        while s != source:
            path_flow = min(path_flow, capacity[parent[s]][s])
            s = parent[s]
        max_flow += path_flow
        v = sink
        while v != source:
            u = parent[v]
            capacity[u][v] -= path_flow
            capacity[v][u] += path_flow
            v = parent[v]
    return max_flow
