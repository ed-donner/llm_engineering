    def is_safe(board, row, col):
        for i in range(row):
            if board[i] == col or \
               board[i] - i == col - row or \
               board[i] + i == col + row:
                return False
        return True

    def solve(board, row):
        if row == n:
            solutions.append(board[:])
            return

            if is_safe(board, row, col):
                board[row] = col
                solve(board, row + 1)

    solutions = []
