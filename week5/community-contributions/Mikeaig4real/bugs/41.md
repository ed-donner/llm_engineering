def prim_mst(graph):
    import heapq
    mst = []
    visited = set()
    min_heap = [(0, 0)]
    
    while min_heap:
        weight, u = heapq.heappop(min_heap)
        if u not in visited:
            visited.add(u)
            mst.append((weight, u))
            for v, w in graph[u]:
                if v not in visited:
                    heapq.heappush(min_heap, (w, v))
    
    return mst[1:]
